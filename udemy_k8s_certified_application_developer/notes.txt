#VS code extension for yaml k8s definition files: YAML
Add below in settings.json:    
    "yaml.schemas": {
        "kubernetes": "*.yaml"
    }

# config maps
kubectl create configmap <configmap_name> --from-literal=<key>=<value> \ 
                                          --from-literal=<key>=<value> 
kubectl create configmap <configmap_name> --from-file=<path_to_file>
#config file format:
APP_COLOR: blue
APP_MODE: prod

kubectl get configmaps
kubectl describe configmaps

#Adding a configmap while creating a pod:
(1) envFrom:
    - configMapRef:
        name: app-config
(2) Single env=>
    env:
    - name: APP_COLOR
      valueFrom:
        configMapKeyRef:
          name: app-config
          key: APP_COLOR

(3) Volumes=>
  volumes:
  - name: app-config-volumes
    configMap:
      name: app-config
